Code Examples

Examples of using the notification system programmatically.

// Initialize notifications (optional - auto-initializes on first use)
notifications.init({
  position: 'top-right',
  defaultDuration: 5000,
  maxNotifications: 5,
  animations: true,
  pauseOnHover: true
});

// Basic notifications
notifications.success('Operation completed successfully!');
notifications.error('Something went wrong!');
notifications.warning('Please check your input');
notifications.info('New update available');

// Advanced notifications
notifications.show({
  type: 'success',
  title: 'Upload Complete',
  message: 'Your file has been uploaded successfully.',
  actions: [
    { text: 'View File', variant: 'primary', action: 'view' },
    { text: 'Share', variant: 'secondary', action: 'share' }
  ],
  duration: 10000,
  data: { fileId: 123 }
});

// Loading notification
const loading = notifications.loading('Processing your request...');
// Later: notifications.update(loading.id, { 
//   type: 'success', 
//   message: 'Request completed!' 
// });

// Alert banner
notifications.banner('Scheduled maintenance tonight from 2-4 AM', 'warning', {
  duration: 10000
});

// Snackbar
notifications.snackbar('Changes saved', {
  action: { 
    text: 'Undo', 
    callback: () => console.log('Undo clicked') 
  }
});

// Event handling
const manager = notifications.init();
manager.on('action', (data) => {
  console.log('Action clicked:', data.action);
});

manager.on('remove', (data) => {
  console.log('Notification removed:', data.notification.id);
});
