Programmatic Button Creation

Examples of creating buttons programmatically using the ButtonBuilder.

// Create buttons with the builder
const primaryBtn = ButtonBuilder.create()
  .variant('primary')
  .size('lg')
  .text('Click Me')
  .icon('üëç', 'left')
  .onClick(() => console.log('Button clicked!'))
  .build();

// Convenience methods
const successBtn = ButtonBuilder.success('Save', () => {
  console.log('Saving...');
});

const dangerBtn = ButtonBuilder.danger('Delete', () => {
  if (confirm('Are you sure?')) {
    console.log('Deleting...');
  }
});

// Button groups with toggle functionality
const group = new ButtonGroup('#my-group', {
  toggle: 'single' // or 'multiple' or 'none'
});

group.on('toggle', (data) => {
  console.log('Active buttons:', data.activeButtons);
});
