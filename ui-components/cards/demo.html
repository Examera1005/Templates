Programmatic Card Creation

Example of creating cards programmatically using the CardBuilder.

// Create a profile card
const profileCard = CardBuilder.create()
  .type('profile')
  .title('John Doe')
  .subtitle('Software Engineer')
  .text('Passionate about creating amazing user experiences')
  .avatar('avatar.jpg', 'John Doe')
  .stats([
    { value: '50+', label: 'Projects' },
    { value: '5', label: 'Years Exp' }
  ])
  .actions([
    { text: 'Connect', variant: 'primary', action: 'connect' },
    { text: 'Message', variant: 'outline', action: 'message' }
  ])
  .build();

// Create a pricing card
const pricingCard = CardBuilder.create()
  .type('pricing')
  .title('Pro Plan')
  .price(29, 'month')
  .featured('Popular')
  .list(['Unlimited projects', '100GB storage', 'Priority support'])
  .actions([{ text: 'Choose Plan', variant: 'primary' }])
  .build();
